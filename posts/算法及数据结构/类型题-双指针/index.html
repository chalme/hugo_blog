无重复字符的最长子串#

滑动窗口；双指针 左右指针都只遍历一次； 时间复杂度: O(n) 空间复杂度: O(∣Σ∣) (出现的字符的次数）

class Solution {
    public int lengthOfLongestSubstring(String s) {
        Set<Character> record = new HashSet<>();

        int ans = 0, n = s.length();
        int j = -1;

        for (int i=0; i<n; i++) {
            if (i != 0) {
                record.remove(s.charAt(i-1));
            }
            while (j+1 <n && !record.contains(s.charAt(j+1))) {
                record.add(s.charAt(j+1));
                j = j+1;
            }
            ans = Math.max(ans, j-i+1);
        }
        return ans;
    }
}

15. 三数之和#

思路 排序 + 双指针； 注意边界

class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        int len = nums.length;
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(nums);
        for (int i=0; i<len; i++) {
            if (i!=0 && nums[i] == nums[i-1]) {
                continue;
            }

            int left = i+1;
            int right = len - 1;
            while (left < right) {
                int three = -(nums[i] + nums[left]);
                while (left < right && nums[right] > three) {
                    right --;
                }
                if (nums[right] == three) {
                    List<Integer> item = new ArrayList<>();
                    item.add(nums[i]);
                    item.add(nums[left]);
                    item.add(nums[right]);
                    result.add(item);
                }
                while (left < right && nums[left+1] == nums[left]) {
                    left ++;
                }
                while (left < right && nums[right-1] == nums[right]) {
                    right --;
                }
                left ++;
            }
        }
        return result;
    }
}

209. 长度最小的子数组#

class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int left = 0, right = 0;
        int sum = 0;
        int res = Integer.MAX_VALUE;
        while (right < n ) {
            sum += nums[right];

            while (sum >= target) {
                if (right - left + 1 < res) {
                    res = right - left + 1;
                }
                sum -= nums[left];
                left ++;
            }
            right ++;
        }
        return res == Integer.MAX_VALUE ? 0 : res;
    }
}

76. 最小覆盖子串#

双指针

class Solution {
    public String minWindow(String s, String t) {

        int[] need = new int[128];
        int[] save = new int[128];
        for (int i=0; i<t.length(); i++) {
            need[t.charAt(i)] ++;
        }
        int len = s.length();
        int min = len + 1;
        int left = 0, right = 0, cnt = 0, start = 0;

        while (right < len) {
            char c = s.charAt(right);
            if (need[c] == 0) {
                right ++;
                continue;
            }
            if (save[c] < need[c]) {
                cnt ++;
            }
            save[c] ++;
            right ++;
            while (cnt == t.length()) {
                if (right - left < min) {
                    min = right - left;
                    start = left;
                }
                char l = s.charAt(left);
                if (need[l] == 0) {
                    left ++;
                    continue;
                }
                if (save[l] == need[l]) {
                    cnt --;
                }
                save[l] --;
                left ++;
            }
        }
        if (min == len + 1) {
            return "";
        }
        return s.substring(start, start + min);
    }
}
